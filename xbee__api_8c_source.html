204 void xbee_ReceiveDataByte(unsigned char c)
205 {
206  char * p;
207 
208  // Wait for the state of a message (do nothing as long as c is the start delimiter)
209  if (xbee_state == XBEE_STATE_DELIMITER)
210  {
211  // printf("\nState Delimiter");
212  if (c == XBEE_DELIMITER)
213  {
214  // Set up next state
215  xbee_rcvPacketSize = 0;
216  xbee_rcvPacketSizeByte = 0;
217  // Go to next state
218  xbee_state = XBEE_STATE_LENGTH;
219  }
220  }
221  // Read the length of the message
222  else if (xbee_state == XBEE_STATE_LENGTH)
223  {
224  // printf("\nState Length");
225  // Assign byte c to size
226  // The first byte receive is the most significant one, the second the least significant one
227  p = (char*)&xbee_rcvPacketSize;
228  p[1 - xbee_rcvPacketSizeByte++] = c;
229 
230  // If size is complete, go to next state
231  if (xbee_rcvPacketSizeByte == XBEE_NUMBER_OF_SIZE_BYTES)
232  {
233  // Set up next state
234  xbee_rcvPacketCurrentByte = 0;
235  xbee_rcvPacketChecksum = 0;
236  // Go to next state
237  xbee_state = XBEE_STATE_API_IDENTIFIER;
238  }
239  }
240  // Read the identifier of the message and go to the dedicated state
241  else if (xbee_state == XBEE_STATE_API_IDENTIFIER)
242  {
243  // printf("\nState API Identifier");
244  if (c == XBEE_IN_MODEM_STATUS)
245  {
246  xbee_state = XBEE_STATE_MODEM_STATUS;
247  }
248  else if (c == XBEE_IN_TX_STATUS)
249  {
250  xbee_state = XBEE_STATE_TX_STATUS;
251  }
252  else if (c == XBEE_IN_AT_COMMAND_RESPONSE)
253  {
254  xbee_atCommandResponse.response = 0;
255  xbee_atCommandResponse.responseSize = 0;
256  xbee_state = XBEE_STATE_AT_RESPONSE;
257  }
258  else if (c == XBEE_IN_RX_64)
259  {
260  xbee_rxFrame64.sourceAddress = 0;
261  xbee_rxFrame64.data = 0;
262  xbee_rxFrame64.dataSize = 0;
263  xbee_state = XBEE_STATE_RX64;
264  }
265  else if (c == XBEE_IN_RX_16)
266  {
267  xbee_rxFrame16.sourceAddress = 0;
268  xbee_rxFrame16.data = 0;
269  xbee_rxFrame16.dataSize = 0;
270  xbee_state = XBEE_STATE_RX16;
271  }
272  else
273  {
274  // Error
275  xbee_state = XBEE_STATE_DELIMITER;
276  // printf("\nERR: API identifier not recognized !");
277  if (c == XBEE_DELIMITER)
278  {
279  // Set up next state
280  xbee_rcvPacketSize = 0;
281  xbee_rcvPacketSizeByte = 0;
282  // Go to next state
283  xbee_state = XBEE_STATE_LENGTH;
284  }
285  }
286 
287  // Save packet type
288  xbee_rcvPacketType = c;
289  // Checksum
290  xbee_rcvPacketChecksum += c;
291 
292  // Start counting the bytes received now
293  ++xbee_rcvPacketCurrentByte;
294  }
295 
296 
297  // Read the content of the modem status message
298  else if (xbee_state == XBEE_STATE_MODEM_STATUS)
299  {
300  // printf("\nState Modem status");
301  // Byte containing status
302  if (xbee_rcvPacketCurrentByte == 1)
303  xbee_modemStatus.status = c;
304 
305  // Checksum
306  xbee_rcvPacketChecksum += c;
307  // Increment number of bytes received
308  ++xbee_rcvPacketCurrentByte;
309 
310  if (xbee_rcvPacketCurrentByte == xbee_rcvPacketSize)
311  xbee_state = XBEE_STATE_CHECKSUM;
312  }
313 
314 
315  // Read the content of the AT response message
316  else if (xbee_state == XBEE_STATE_AT_RESPONSE)
317  {
318  // printf("\nState AT response");
319  // frame id
320  if (xbee_rcvPacketCurrentByte == 1)
321  {
322  xbee_atCommandResponse.frameId = c;
323  }
324  // first character of AT command
325  else if (xbee_rcvPacketCurrentByte == 2)
326  {
327  xbee_atCommandResponse.command[0] = c;
328  }
329  // second character of AT command
330  else if (xbee_rcvPacketCurrentByte == 3)
331  {
332  xbee_atCommandResponse.command[1] = c;
333  }
334  // status of the command
335  else if (xbee_rcvPacketCurrentByte == 4)
336  {
337  xbee_atCommandResponse.status = c;
338  }
339  // first byte of the parameter value
340  else if (xbee_rcvPacketCurrentByte == 5)
341  {
342  xbee_atCommandResponse.response = xbee_rcvBuffer;
343  xbee_rcvBuffer[xbee_atCommandResponse.responseSize++] = c;
344  }
345  // following bytes of the parameter value
346  else
347  {
348  xbee_rcvBuffer[xbee_atCommandResponse.responseSize++] = c;
349  }
350 
351  // Checksum
352  xbee_rcvPacketChecksum += c;
353  // Increment number of bytes received
354  ++xbee_rcvPacketCurrentByte;
355 
356  if (xbee_rcvPacketCurrentByte == xbee_rcvPacketSize)
357  xbee_state = XBEE_STATE_CHECKSUM;
358  }
359 
360  // Read the content of the TX status message
361  else if (xbee_state == XBEE_STATE_TX_STATUS)
362  {
363  // printf("\nState TX status");
364  if (xbee_rcvPacketCurrentByte == 1)
365  {
366  xbee_txStatus.frameId = c;
367  }
368  else if (xbee_rcvPacketCurrentByte == 2)
369  {
370  xbee_txStatus.status = c;
371  }
372 
373  // Checksum
374  xbee_rcvPacketChecksum += c;
375  // Increment number of bytes received
376  ++xbee_rcvPacketCurrentByte;
377 
378 
379  if (xbee_rcvPacketCurrentByte == xbee_rcvPacketSize)
380  xbee_state = XBEE_STATE_CHECKSUM;
381  }
382 
383 
384  // Read the content of the incoming RF message with a 64-bits address
385  else if (xbee_state == XBEE_STATE_RX64)
386  {
387  // printf("\nState RX64");
388  if (xbee_rcvPacketCurrentByte >= 1 && xbee_rcvPacketCurrentByte <= 8)
389  {
390  p = (char*)&(xbee_rxFrame64.sourceAddress);
391  p[7 - (xbee_rcvPacketCurrentByte - 1)] = c;
392  }
393  else if (xbee_rcvPacketCurrentByte == 9)
394  {
395  xbee_rxFrame64.RSSI = c;
396  }
397  else if (xbee_rcvPacketCurrentByte == 10)
398  {
399  xbee_rxFrame64.options = c;
400  }
401  else if (xbee_rcvPacketCurrentByte == 11)
402  {
403  xbee_rxFrame64.data = xbee_rcvBuffer;
404  }
405 
406  if (xbee_rcvPacketCurrentByte >= 11)
407  {
408  xbee_rcvBuffer[xbee_rxFrame64.dataSize++] = c;
409 
410  // Give this byte further
411  if (xbee_rcvRfDataByteFunc)
412  xbee_rcvRfDataByteFunc(c);
413  }
414 
415  // Checksum
416  xbee_rcvPacketChecksum += c;
417  // Increment number of bytes received
418  ++xbee_rcvPacketCurrentByte;
419 
420 
421  if (xbee_rcvPacketCurrentByte == xbee_rcvPacketSize)
422  xbee_state = XBEE_STATE_CHECKSUM;
423  }
424 
425 
426  // Read the content of the incoming RF message with a 16-bits address
427  else if (xbee_state == XBEE_STATE_RX16)
428  {
429  // printf("\nState RX16");
430  // printf("\n...byte %u",xbee_rcvPacketCurrentByte);
431 
432  if (xbee_rcvPacketCurrentByte >= 1 && xbee_rcvPacketCurrentByte <= 2)
433  {
434  p = (char*)&(xbee_rxFrame16.sourceAddress);
435  p[1 - (xbee_rcvPacketCurrentByte - 1)] = c;
436  }
437  else if (xbee_rcvPacketCurrentByte == 3)
438  {
439  xbee_rxFrame16.RSSI = c;
440  }
441  else if (xbee_rcvPacketCurrentByte == 4)
442  {
443  xbee_rxFrame16.options = c;
444  }
445  else if (xbee_rcvPacketCurrentByte == 5)
446  {
447  xbee_rxFrame16.data = xbee_rcvBuffer;
448  }
449 
450  if (xbee_rcvPacketCurrentByte >= 5)
451  {
452  xbee_rcvBuffer[xbee_rxFrame16.dataSize++] = c;
453 
454  // Give this byte further
455  if (xbee_rcvRfDataByteFunc)
456  xbee_rcvRfDataByteFunc(c);
457  }
458 
459  // Checksum
460  xbee_rcvPacketChecksum += c;
461  // Increment number of bytes received
462  ++xbee_rcvPacketCurrentByte;
463 
464  if (xbee_rcvPacketCurrentByte == xbee_rcvPacketSize)
465  xbee_state = XBEE_STATE_CHECKSUM;
466  }
467 
468 
469  // Read the checksum of the message
470  else if (xbee_state == XBEE_STATE_CHECKSUM)
471  {
472  // printf("\nState checksum");
473  // Checksum
474  xbee_rcvPacketChecksum += c;
475 
476  // printf("\nchecksum %X %X",xbee_rcvPacketChecksum, c);
477 
478 
479  // Message is valid
480  if (xbee_rcvPacketChecksum == 0xFF)
481  {
482  // printf("\nPacket valid, type %X", xbee_rcvPacketType);
483 
484  if (xbee_rcvPacketType == XBEE_IN_MODEM_STATUS)
485  {
486  if (xbee_rcvModemStatusFunc)
487  xbee_rcvModemStatusFunc(&xbee_modemStatus);
488  }
489  else if (xbee_rcvPacketType == XBEE_IN_TX_STATUS)
490  {
491  if (xbee_rcvTxStatusFunc)
492  xbee_rcvTxStatusFunc(&xbee_txStatus);
493  }
494  else if (xbee_rcvPacketType == XBEE_IN_AT_COMMAND_RESPONSE)
495  {
496  if (xbee_rcvAtCommandResponseFunc)
497  xbee_rcvAtCommandResponseFunc(&xbee_atCommandResponse);
498  }
499  else if (xbee_rcvPacketType == XBEE_IN_RX_64)
500  {
501  if (xbee_rcvRfData64Func)
502  xbee_rcvRfData64Func(&xbee_rxFrame64);
503  }
504  else if (xbee_rcvPacketType == XBEE_IN_RX_16)
505  {
506  if (xbee_rcvRfData16Func)
507  xbee_rcvRfData16Func(&xbee_rxFrame16);
508  }
509 
510  xbee_state = XBEE_STATE_DELIMITER;
511  }
512  // Message is not valid
513  else
514  {
515  // printf("\nERR: Packet NOT valid");
516  // Do nothing, drop it !
517  if (c == XBEE_DELIMITER)
518  {
519  // Set up next state
520  xbee_rcvPacketSize = 0;
521  xbee_rcvPacketSizeByte = 0;
522  // Go to next state
523  xbee_state = XBEE_STATE_LENGTH;
524  }
525  else
526  {
527  xbee_state = XBEE_STATE_DELIMITER;
528  }
529  }
530  }
531  else
532  {
533  // Error, invalid state, should never happen
534  xbee_state = XBEE_STATE_DELIMITER;
535  }
536  // flush(stdout);
537 }

545 void xbee_SendTxFrame64(xbee_TxFrame64 * packet)
546 {
547  // Add by Mickael Germain LIRMM 09/07/2014 : change from 5 to 11 (problem with address size between 16 and 64 bits).
548  // unsigned short int size = 5 + packet->dataSize;
549  unsigned short int size = 11 + packet->dataSize;
550  unsigned char delimiter = XBEE_DELIMITER;
551  unsigned char identifier = XBEE_OUT_TX_64;
552  unsigned char checksum = 0;
553  unsigned short int i = 0;
554 
555  checksum += identifier;
556  checksum += packet->frameId;
557  for (i = 0; i < 8; ++i)
558  {
559  checksum += *(((char*)&(packet->destinationAddress))+i);
560  }
561  checksum += packet->options;
562  for (i = 0; i < packet->dataSize; ++i)
563  {
564  checksum += packet->data[i];
565  }
566  checksum = (0xFF - checksum) & 0xFF;
567 
568  xbee_SendBytes((unsigned char*)&delimiter, 1);
569  xbee_SendBytes(((unsigned char*)&size) + 1, 1);
570  xbee_SendBytes(((unsigned char*)&size), 1);
571  xbee_SendBytes((unsigned char*)&identifier, 1);
572  xbee_SendBytes((unsigned char*)&(packet->frameId), 1);
573  for (i = 0; i < 8; ++i)
574  {
575  xbee_SendBytes((((unsigned char*)&(packet->destinationAddress)) + (7 - i)), 1);
576  }
577  xbee_SendBytes((unsigned char*)&(packet->options), 1);
578  if (packet->dataSize > 0 && packet->data)
579  xbee_SendBytes(packet->data, packet->dataSize);
580  xbee_SendBytes((unsigned char*)&checksum, 1);
581 
582  if (xbee_flushFunc)
583  xbee_flushFunc();
584 }

729 void xbee_FlushPacket()
730 {
731  // set frame id
732  ++xbee_sndPacketFrameId;
733  if (xbee_sndPacketFrameId == 0)
734  {
735  xbee_sndPacketFrameId = 1; // not 0 because 0 is a special number that disables TX status responses
736  }
737  xbee_txFrame16.frameId = xbee_sndPacketFrameId;
738 
739  // link data to frame
740  xbee_txFrame16.data = xbee_sndBuffer;
741  xbee_txFrame16.dataSize = xbee_sndPacketSize;
742 
743  // send frame
744  xbee_SendTxFrame16(&xbee_txFrame16);
745 
746  // printf("\nPacket sent : %u",xbee_sndPacketSize);
747 
748  // reset size for next packet
749  xbee_sndPacketSize = 0;
750 }
