22  assert(nbData <= _24XX1026_PAGE_SIZE); /* Always true thanks to uint8_t. */
23  /* Write Byte :
24  * MSB|6|5|4| 3| 2|1|LSB
25  * 1|0|1|0|A1|A0|B|R/W
26  *
27  * Omit the LSB

97  */
